import java.util.*;
public class Main
{
    public static void main(String args[])throws java.lang.Exception
    {
       Scanner sc= new Scanner(System.in);
       int i=0;
       int t = sc.nextInt();
       while(i<t)
       {
           
           int a=sc.nextInt();
           int b = sc.nextInt();
           int c = sc.nextInt();
           if(a>0 && b>0 && c>0 && (a+b+c)==180 )
               System.out.println("YES");
           else
               System.out.println("NO");
        i++;
       }

       sc.close();
    }
}
