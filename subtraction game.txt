        {

            int arr[]=new int[a];
            int ans=0;
            for(int i=0;i<a;i++)

                arr[i]=sc.nextInt();
                
            }
            int temp=gcd(arr[0],arr[1]);
            for(int i=3;i<=a;i++)
            {
                temp=gcd(temp,arr[i-1]);

    static int gcd(int c,int d)
    {
        if(d==0)
            return c;
        else
            return gcd(d,c%d);
        
    }
}
